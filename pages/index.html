def usr_request():
    print("Welcome to Corder! Please enter your coding request:")
    request = input()
    if request.strip() != "":
        if request.lower() == "exit":
            print("Thank you for using Corder! Goodbye!")
            exit()
        elif request.lower() == "help":
            print("Corder is a platform that supports your creativity and imagination. You can ask for coding help, project ideas, or anything related to programming. Just type your request and we'll do our best to assist you!")
        elif request.lower() == "about":
            print("Corder is a team of passionate individuals dedicated to providing a platform for creative expression and imagination. Our mission is to support and empower creators from all walks of life, fostering a community where ideas can flourish and creativity can thrive. Whether you're an artist, writer, musician, or any type of creator, we are here to help you share your work with the world and connect with like-minded individuals who share your passion. Join us on this journey of creativity and let's build something amazing together!")
        elif request.lower() == "languages":
            print("Corder supports a wide range of programming languages, including Python, C++, Lua, JavaScript, Java, C#, Go, Rust, Ruby, PHP, Swift, Kotlin, TypeScript, and Scala. If you have a specific language in mind that you'd like to work with, just let us know and we'll do our best to assist you!")
        else:
            print("Thank you for your request! We will get back to you as soon as possible with the information you need.")            
    else:
        print("Please enter a valid request.")
        return usr_request()
